[김영삼] [오후 8:37] import numpy as np
import pandas as pd
import yfinance as yf
import matplotlib.pyplot as plt

# 데이터 다운로드
ticker = 'SPY'
data = yf.download(ticker, start='2000-01-01')
data['Return'] = data['Close'].pct_change()

# 모멘텀 지표 계산 (예: 10일 이동평균 수익률)
data['Momentum'] = data['Return'].rolling(window=10).mean()

# 매매 신호 생성
threshold_up = 0.01  # 매수 신호 임계값
threshold_down = -0.01  # 매도 신호 임계값
data['Signal'] = 0.0
data['Signal'] = np.where(data['Momentum'] > threshold_up, 1.0, data['Signal'])
data['Signal'] = np.where(data['Momentum'] < threshold_down, -1.0, data['Signal'])
data['Position'] = data['Signal'].shift(1)

# 포트폴리오 생성
initial_capital = 100000.0
positions = pd.DataFrame(index=data.index).fillna(0.0)
positions[ticker] = data['Position']
portfolio = positions.multiply(data['Close'], axis=0)
pos_diff = positions.diff()
portfolio['holdings'] = (positions.multiply(data['Close'], axis=0)).sum(axis=1)
portfolio['cash'] = initial_capital - (pos_diff.multiply(data['Close'], axis=0)).sum(axis=1).cumsum()
portfolio['total'] = portfolio['cash'] + portfolio['holdings']

# 성능 지표 계산
portfolio['returns'] = portfolio['total'].pct_change()
CAGR = (portfolio['total'][-1] / portfolio['total'][1]) ** (1/(len(portfolio)/252)) - 1
MDD = ((portfolio['total']/portfolio['total'].cummax()) - 1).min()

print(f"연평균수익률(CAGR): {CAGR * 100:.2f}%")
print(f"최대하락률(MDD): {MDD * 100:.2f}%")

# 그래프 그리기
plt.figure(figsize=(12, 6))
plt.plot(data['Close'], label='SPY')
plt.plot(portfolio['total'], label='Portfolio value')
plt.legend(loc='best')
plt.title('S&P 500 Enhanced Trading Strategy Performance')
plt.show()
